function calculateWork() { const startDate = new Date(); const endDate = new Date('November 17, 2023 23:59:59'); if (startDate > endDate) { alert('A data de início é posterior à data de término. Por favor, ajuste as datas.'); return; } document.getElementById('startDate').textContent = startDate.toLocaleDateString(); const totalWorkDays = countWorkdays(startDate, endDate); document.getElementById('workdays').textContent = totalWorkDays; if (totalWorkDays === 0) { document.getElementById('hoursPerWeek').textContent = 'N/A - Não há dias úteis no intervalo'; return; } const weeks = totalWorkDays / 5; const hoursPerWeek = (150 / weeks).toFixed(2); document.getElementById('hoursPerWeek').textContent = hoursPerWeek; } function countWorkdays(start, end) { let count = 0; const day = new Date(start); while (day < end) { const dayOfWeek = day.getDay(); if (dayOfWeek >= 1 && dayOfWeek <= 5) count++; day.setDate(day.getDate() + 1); } return count; }
